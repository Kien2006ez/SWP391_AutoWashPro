\subsection{Sơ đồ Use Case (Use Case Diagram)}

\subsubsection{Mô tả tổng quan hệ thống}
Hệ thống Quản lý Rửa xe Tự động Thông minh (AutoWashPro) bao gồm 3 tác nhân chính tương tác trực tiếp với hệ thống:
\begin{itemize}
    \item \textbf{Khách hàng (Customer):} Thực hiện đăng ký, đăng nhập, đặt lịch rửa xe trực tuyến và theo dõi lịch sử tích điểm.
    \item \textbf{Nhân viên (Staff):} Thực hiện đăng nhập để cập nhật trạng thái tiến độ rửa xe và xử lý thanh toán cho khách (có thể áp dụng mã giảm giá).
    \item \textbf{Quản trị viên (Admin):} Toàn quyền quản lý nhân viên, cấu hình các gói dịch vụ rửa xe và xem báo cáo thống kê doanh thu.
\end{itemize}

\subsubsection{Đặc tả chi tiết một số Use Case chính}

\paragraph{Use Case: Đặt lịch rửa xe}
\begin{itemize}
    \item \textbf{Actor chính:} Khách hàng.
    \item \textbf{Mục tiêu:} Giúp khách hàng đặt lịch hẹn trước nhằm tiết kiệm thời gian chờ đợi.
    \item \textbf{Luồng sự kiện:} Khách hàng chọn thời gian, chi nhánh $\rightarrow$ Hệ thống tính tiền $\rightarrow$ Lưu lịch hẹn thành công.
\end{itemize}

\paragraph{Use Case: Cập nhật trạng thái xe}
\begin{itemize}
    \item \textbf{Actor chính:} Nhân viên.
    \item \textbf{Mục tiêu:} Chuyển trạng thái xe (Đang chờ $\rightarrow$ Đang rửa $\rightarrow$ Đã xong) để khách hàng theo dõi real-time.
\end{itemize}

\paragraph{Use Case: Xem báo cáo doanh thu}
\begin{itemize}
    \item \textbf{Actor chính:} Quản trị viên.
    \item \textbf{Mục tiêu:} Thống kê số lượng xe đã rửa và doanh thu theo ngày, tháng, năm để chủ cửa hàng nắm bắt tình hình kinh doanh.
\end{itemize}
