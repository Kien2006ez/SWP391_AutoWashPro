\chapter{Tổng Quan Về Dự Án Rửa Xe Thông Minh}

\section{Lý do chọn đề tài}
Trong bối cảnh công nghệ phát triển mạnh mẽ, việc tối ưu hóa dịch vụ chăm sóc xe thông qua các ứng dụng trực tuyến là xu hướng tất yếu. Dự án Hệ thống Quản lý Rửa xe Thông minh (\textbf{AutoWashPro}) được phát triển nhằm giải quyết triệt để các vấn đề về thời gian chờ đợi của khách hàng, tối ưu hóa quy trình quản lý vận hành cho các trung tâm dịch vụ, đồng thời ứng dụng các mô hình tích lũy điểm thưởng thành viên hiện đại nhằm nâng cao tỷ lệ giữ chân khách hàng (Customer Retention Rate Rate) phục vụ nghiên cứu khoa học.

\section{Mục tiêu của dự án}
Hệ thống hướng tới việc cung cấp một giải pháp toàn diện bao gồm:
\begin{itemize}
    \item Cho phép khách hàng chủ động quản lý phương tiện, đặt lịch rửa xe trực tuyến, chọn các gói dịch vụ và tham gia vào hệ sinh thái tích đổi điểm thưởng thành viên linh hoạt.
    \item Cung cấp bảng điều khiển quản trị (Admin Dashboard) tối ưu cho nhân viên và quản lý trung tâm trong việc tiếp nhận xe, xuất hóa đơn nội bộ, điều phối lịch hẹn và theo dõi các chỉ số tăng trưởng doanh thu theo thời gian thực.
\end{itemize}

\section{Phạm vi hệ thống và Các phân hệ chức năng (Jira Project: KAN)}
Dự án tập trung nghiên cứu, xây dựng và hiện thực hóa các khối chức năng chính được quản lý theo mô hình Agile/Scrum trên Jira Board dự án \textbf{KAN}. Mỗi thẻ công việc (Task) được gắn nhãn (\textit{Label}) song song theo \textbf{Feature} (Feature-1 đến Feature-7) và theo \textbf{thành viên phụ trách} (M1 đến M7), cụ thể bao gồm 7 phân hệ sau:

\begin{enumerate}
    \item \textbf{Xác thực \& Quản lý Tài khoản (Feature 1):}
    \begin{itemize}
        \item Xây dựng cơ sở dữ liệu người dùng, thực hiện mã hóa mật khẩu một chiều bằng thuật toán bcrypt.
        \item API \texttt{POST /auth/register} và \texttt{POST /auth/login} kèm middleware xác thực JWT (Access Token) bảo vệ các endpoint riêng tư.
        \item Giao diện Đăng ký / Đăng nhập phía khách hàng và Admin.
    \end{itemize}

    \item \textbf{Quản lý Hồ sơ xe (Feature 2):}
    \begin{itemize}
        \item Khách hàng quản lý danh sách phương tiện cá nhân (thêm mới, cập nhật biển số, hãng xe, dòng xe).
        \item CRUD API cho phương tiện, ràng buộc quyền sở hữu (chỉ chủ xe mới sửa/xóa được hồ sơ xe của mình) và kiểm tra định dạng biển số.
    \end{itemize}

    \item \textbf{Đặt lịch dịch vụ (Booking) (Feature 3):}
    \begin{itemize}
        \item API tạo lịch hẹn, tra cứu lịch sử đặt lịch và hủy lịch hẹn theo trạng thái.
        \item Tích hợp logic kiểm tra khung giờ trống (Slot) theo thời gian thực, đối chiếu với phân hạng thành viên để áp quy tắc cửa sổ thời gian đặt lịch trước (\textit{Booking Window}: Member 7 ngày, Silver 10 ngày, Gold 12 ngày, Platinum 14 ngày) và hàng đợi ưu tiên theo hạng thành viên.
    \end{itemize}

    \item \textbf{Thành viên \& Điểm thưởng Loyalty (Feature 4):}
    \begin{itemize}
        \item Bộ máy tính điểm (\textit{Point Calculation Engine}) ghi nhận điểm thưởng (\textit{Wash Credits}) từ các lịch hẹn đã hoàn thành, nhân theo hệ số (\textit{Multiplier}) tương ứng cấp hạng (Member, Silver, Gold, Platinum).
        \item Cơ chế đổi điểm (\textit{Redemption}) lấy discount/free wash/add-on và điểm tự động hết hạn sau 12 tháng.
        \item Cron Job chạy định kỳ đầu tháng để tự động nâng/hạ hạng thành viên dựa trên chi tiêu tích lũy.
    \end{itemize}

    \item \textbf{Khuyến mãi \& Mã giảm giá (Promotion) (Feature 5):}
    \begin{itemize}
        \item CRUD API quản lý chương trình khuyến mãi và mã giảm giá.
        \item Logic áp dụng khuyến mãi theo đối tượng mục tiêu (\textit{Target Tier}, ví dụ chỉ gửi cho "Silver trở lên") và theo khung thời gian hiệu lực.
    \end{itemize}

    \item \textbf{Quản trị \& Điều phối Trung tâm cho Admin/Staff (Feature 6):}
    \begin{itemize}
        \item Dashboard thống kê doanh thu, số lịch hẹn, hoạt động khách hàng theo thời gian thực.
        \item API quản lý tài khoản khách hàng và quản lý/điều phối trạng thái lịch hẹn (Pending, Confirmed, Completed, Cancelled) tập trung cho nhân viên tại quầy.
    \end{itemize}

    \item \textbf{Khảo sát \& Nghiên cứu dữ liệu (Survey \& Research) (Feature 7):}
    \begin{itemize}
        \item API tiếp nhận và lưu trữ phản hồi khảo sát khách hàng phục vụ mục tiêu nghiên cứu (RQ).
        \item Bộ trích xuất dữ liệu hành vi (\textit{Research Logger}) xuất ra tệp phẳng \texttt{.csv} phục vụ phân tích thống kê/huấn luyện mô hình.
    \end{itemize}
\end{enumerate}
