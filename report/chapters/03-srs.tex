\chapter{Phân Tích Yêu Cầu Hệ Thống (SRS)}

\section{Tổng quan hệ thống và các ràng buộc (System Overview \& Constraints)}
Hệ thống vận hành trên môi trường mạng Internet, Client truy cập thông qua trình duyệt web responsive hiện đại (Chrome, Safari, Edge). Phía Back-end xử lý các luồng nghiệp vụ tập trung kết nối cơ sở dữ liệu quan hệ.

Cấu trúc lõi của hệ thống dựa trên Bảng đặc quyền phân hạng thành viên (Multi-Tier Loyalty Structure) được quy định nghiêm ngặt như sau:
\begin{longtable}{|p{2.5cm}|c|c|p{2.5cm}|p{4.5cm}|}
\hline
\textbf{Phân hạng (Tier)} & \textbf{Chi tiêu tối thiểu} & \textbf{Hệ số điểm} & \textbf{Booking Window} & \textbf{Đặc quyền ưu tiên} \\ \hline
\textbf{Member} & 0 VNĐ & 1.0x & Tối đa 7 ngày trước & Đặt lịch cơ bản. \\ \hline
\textbf{Silver} & 500,000 VNĐ & 1.2x & Tối đa 10 ngày trước & Đặt lịch trước nhóm Member, giảm 5\% ngày sinh nhật. \\ \hline
\textbf{Gold} & 1,500,000 VNĐ & 1.5x & Tối đa 12 ngày trước & Lối xếp hàng ưu tiên tại quầy, quà tặng sinh nhật đặc biệt. \\ \hline
\textbf{Platinum} & 4,000,000 VNĐ & 2.0x & Tối đa 14 ngày trước & Đặt slot giờ cao điểm cố định, miễn phí dịch vụ đi kèm. \\ \hline
\end{longtable}

\noindent \textbf{Các ràng buộc hệ thống (System Constraints):}
\begin{itemize}
    \item Hệ thống không tích hợp trực tiếp cổng thanh toán trực tuyến tự động cho khách hàng. Quy trình kiểm soát trạng thái thanh toán và kích hoạt tích điểm hoàn toàn do nhân viên quầy (Admin) xác nhận thủ công trên hệ thống sau khi khách hàng hoàn thành dịch vụ tại cửa hàng.
    \item Quy trình nâng hạ hạng được quét tự động bởi hệ thống vào ngày đầu tiên của mỗi tháng dựa trên tổng chi tiêu tích lũy của 3 tháng liền kề trước đó.
\end{itemize}

\section{Tác nhân và Sơ đồ Use Case (Actors \& Use Cases)}
Hệ thống tương tác trực tiếp với hai tác nhân chính (Primary Actors):
\begin{itemize}
    \item \textbf{Customer (Khách hàng):} Người sở hữu phương tiện, thực hiện các hành động đăng ký, quản lý danh sách biển số xe cá nhân, tra cứu thông tin phân hạng, đặt lịch hẹn rửa xe theo khung giờ khả dụng và thực hiện đổi các mã phần thưởng từ ví điểm tích lũy (\textit{Wash Credits}).
    \item \textbf{Admin (Chủ cơ sở / Nhân viên quầy):} Người vận hành hệ thống, thực hiện kiểm tra xe vào bãi, bấm xác nhận hoàn thành dịch vụ rửa xe để kích hoạt hóa đơn tích điểm, cấu hình các chương trình khuyến mãi theo hạng mục tiêu và xuất dữ liệu hành vi.
\end{itemize}

\section{Yêu cầu chức năng chi tiết (Functional Requirements)}
Các chức năng bắt buộc phải được triển khai chi tiết theo 3 Phân hệ cốt lõi (Features):

\subsection{Feature 1: Đăng nhập \& Đăng ký tài khoản}
\textit{Thành viên phụ trách: Lê Thị Quỳnh Nhung (NQ - BA), Trần Tiến Dũng (DT - Backend), Lê Sỹ Nghĩa (T - Frontend)}

Phân hệ này cung cấp cơ chế xác thực an toàn cho toàn bộ người dùng của hệ thống. Bao gồm các chức năng:
\begin{enumerate}
    \item \textbf{[F1-DB] Database Schema:} Thiết kế bảng \textit{customers} và bảng \textit{otp\_verifications} phục vụ lưu trữ thông tin định danh và lịch sử mã khóa an toàn với index tối ưu hóa.
    \item \textbf{[F1-BE-1] API Register:} Endpoint \texttt{POST /api/auth/register} tiếp nhận thông tin người dùng. Thực hiện kiểm tra định dạng email và độ mạnh mật khẩu (tối thiểu 8 ký tự, có chữ hoa, chữ thường và số). Thực hiện mã hóa mật khẩu bằng thuật toán Bcrypt (cost factor \(\geq\) 12) trước khi gửi OTP xác thực.
    \item \textbf{[F1-BE-2] API Verify OTP:} Endpoint \texttt{POST /api/auth/verify-otp} tiếp nhận mã kích hoạt 6 số. Kiểm tra thời hạn hiệu lực (5 phút) để kích hoạt tài khoản sang trạng thái \textit{Active}.
    \item \textbf{[F1-BE-3] API Login:} Endpoint \texttt{POST /api/auth/login} xác thực thông tin đăng nhập, sinh chuỗi khóa JSON Web Token (bao gồm \textit{access\_token} và \textit{refresh\_token}).
    \item \textbf{[F1-FE-1] Giao diện đăng ký và xác thực:} Xây dựng biểu mẫu đăng ký thời gian thực và màn hình đếm ngược 5 phút nhập mã OTP kèm tính năng gửi lại mã (\textit{Resend OTP}).
\end{enumerate}

\subsection{Feature 2: Quản lý phương tiện \& Đặt lịch rửa xe}
\textit{Thành viên phụ trách: Phan Triều Cường (LK - Database), Trần Tiến Dũng (DT - Backend), Lê Sỹ Nghĩa (T - Frontend)}

Phân hệ này cho phép khách hàng quản lý thông tin phương tiện cá nhân và đặt trước lịch hẹn dịch vụ:
\begin{enumerate}
    \item \textbf{[F2-DB-1] Schema thiết kế cấu trúc hình học:} Thiết lập mối quan hệ giữa các bảng \textit{customer\_vehicles}, \textit{bookings} và bảng danh mục gói dịch vụ \textit{services}.
    \item \textbf{[F2-BE-1] API Lấy thông tin danh mục:} Endpoint \texttt{GET /api/services} cung cấp danh mục dịch vụ kèm bộ lọc và phân trang dữ liệu.
    \item \textbf{[F2-BE-2] API Kiểm tra khung giờ trống:} Endpoint \texttt{GET /api/bookings/available-slots} tính toán thời gian khả dụng theo thời gian thực dựa trên thời lượng triển khai của gói dịch vụ đã chọn và các lịch đặt thành công trước đó.
    \item \textbf{[F2-BE-3] API Khởi tạo lịch hẹn:} Endpoint \texttt{POST /api/bookings} ràng buộc điều kiện đăng nhập, kiểm tra quy tắc cửa sổ thời gian (\textit{Booking Window}) tương ứng với phân hạng thành viên của khách hàng hiện tại trước khi thiết lập trạng thái \textit{Pending}.
    \item \textbf{[F2-FE-1] Giao diện thiết lập đặt lịch:} Triển khai Widget Lịch thông minh hỗ trợ hiển thị trực quan các khung giờ trống và màn hình quản lý hồ sơ xe (thêm, sửa, xóa biển số xe).
\end{enumerate}

\subsection{Feature 3: Quản lý hóa đơn nội bộ \& Tích điểm thành viên}
\textit{Thành viên phụ trách: Phan Triều Cường (LK - Database), Trần Tiến Dũng (DT - Backend)}

Phân hệ xử lý dữ liệu tài chính nội bộ, tính toán nâng hạng và ghi nhận đổi thưởng:
\begin{enumerate}
    \item \textbf{[F3-DB-1] Schema bảng dữ liệu tài chính:} Thiết lập bảng \textit{bookings}, \textit{customer\_points} (quản lý ví Wash Credits) và bảng \textit{system\_promotions}.
    \item \textbf{[F3-BE-1] Cơ chế tự động khởi tạo hóa đơn:} Khi nhân viên quầy kiểm tra xe và click chuyển trạng thái lịch hẹn thành \textit{Completed}, hệ thống tự động kết xuất dữ liệu tính toán tổng tiền bằng công thức cơ bản: \(\text{Total} = \text{Price} + \text{Tax (10\% Price)}\).
    \item \textbf{[F3-BE-2] API Quản lý ví điểm và Đổi thưởng:} Ghi nhận cộng điểm thưởng vào ví của khách hàng theo đúng hệ số điểm (\textit{Multiplier}) của hạng thành viên hiện tại. Đồng thời xử lý trừ điểm khi gọi lệnh đổi mã ưu đãi.
    \item \textbf{[F3-BE-3] API Quét hạng tự động (Cron Job):} Tác vụ chạy ngầm được lập lịch kích hoạt vào lúc 00:00 ngày đầu tháng, tự động quét tổng chi tiêu trong 3 tháng gần nhất để thực thi lệnh thăng/hạ hạng người dùng.
    \item \textbf{[F3-FE-1] Bảng điều khiển quản trị (Admin Dashboard):} Giao diện hiển thị biểu đồ xu hướng doanh thu và danh sách hóa đơn chờ xử lý dành riêng cho người vận hành tại quầy dịch vụ.
\end{enumerate}

\noindent \textbf{Tóm tắt số lượng công việc (Jira Tasks):} Phân hệ Feature 1 gồm 11 nhiệm vụ, Feature 2 gồm 16 nhiệm vụ, Phân hệ Feature 3 gồm 18 nhiệm vụ. Tổng cộng 51 đầu việc lập trình và đặc tả được kiểm soát chặt chẽ trên bảng quản trị Jira (mã thẻ công việc từ KAN-45 đến KAN-95).

\section{Yêu cầu phi chức năng (Non-Functional Requirements)}
\begin{itemize}
    \item \textbf{Hiệu năng (Performance):} Thời gian phản hồi xử lý tính toán rẽ nhánh của các API đọc dữ liệu phải nhỏ hơn 1 giây dưới áp lực 100 người dùng truy cập đồng thời (\textit{Concurrent Users}). Tốc độ phản hồi giao diện Frontend không quá 3 giây.
    \item \textbf{Bảo mật (Security):} Toàn bộ mật khẩu lưu trữ bắt buộc đi qua bộ băm Bcrypt (\textit{cost factor} \(\geq\) 12). Phiên làm việc tương tác qua API được xác thực bằng chữ ký JSON Web Token (JWT) lưu an toàn trong thiết lập Cookie bảo mật nhằm ngăn chặn hoàn toàn lỗ hổng tấn công XSS.
    \item \textbf{Tính tin cậy và Nhất quán (Reliability):} Tác vụ xử lý tự động đầu tháng (Cron Job) phải tích hợp cơ chế \textit{Idempotency} (đảm bảo tính duy nhất), tuyệt đối không xảy ra hiện tượng chạy trùng lặp dữ liệu gây sai sót điểm số thành viên.
\end{itemize}

\section{Các quy tắc nghiệp vụ cốt lõi (Business Rules)}
Hệ thống áp dụng chặt chẽ bộ quy tắc nghiệp vụ bao gồm: Quy tắc xác thực thông tin tài khoản (\textit{Identity Rules}), Quy tắc giới hạn thời gian và hủy lịch hẹn (\textit{Booking Rules}), Quy tắc tính toán thăng hạng và hết hạn điểm (\textit{Loyalty Rules}), Quy tắc kiểm tra điều kiện áp dụng mã giảm giá (\textit{Promotion Rules}), Quy tắc bảo mật phân quyền dữ liệu (\textit{Security Rules}) và Quy tắc tách biệt dữ liệu định danh trong phân tích phục vụ nghiên cứu khoa học (\textit{Research Rules}).
