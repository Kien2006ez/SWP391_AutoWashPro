\chapter{Phân Tích Yêu Cầu Hệ Thống (SRS)}

\section{Tổng quan hệ thống và các ràng buộc (System Overview \& Constraints)}
Hệ thống vận hành trên môi trường mạng Internet, Client truy cập thông qua trình duyệt web responsive hiện đại (Chrome, Safari, Edge). Phía Back-end xử lý các luồng nghiệp vụ tập trung kết nối cơ sở dữ liệu quan hệ.

Cấu trúc lõi của hệ thống dựa trên Bảng đặc quyền phân hạng thành viên (Multi-Tier Loyalty Structure) được quy định nghiêm ngặt như sau:
\begin{longtable}{|p{2.5cm}|c|c|p{2.5cm}|p{4.5cm}|}
\hline
\textbf{Phân hạng (Tier)} & \textbf{Chi tiêu tối thiểu} & \textbf{Hệ số điểm} & \textbf{Booking Window} & \textbf{Đặc quyền ưu tiên} \\ \hline
\textbf{Member} & 0 VNĐ & 1.0x & Tối đa 7 ngày trước & Đặt lịch cơ bản. \\ \hline
\textbf{Silver} & 500,000 VNĐ & 1.2x & Tối đa 10 ngày trước & Đặt lịch trước nhóm Member, giảm 5\% ngày sinh nhật. \\ \hline
\textbf{Gold} & 1,500,000 VNĐ & 1.5x & Tối đa 12 ngày trước & Lối xếp hàng ưu tiên tại quầy, quà tặng sinh nhật đặc biệt. \\ \hline
\textbf{Platinum} & 4,000,000 VNĐ & 2.0x & Tối đa 14 ngày trước & Đặt slot giờ cao điểm cố định, miễn phí dịch vụ đi kèm. \\ \hline
\end{longtable}

\noindent \textbf{Các ràng buộc hệ thống (System Constraints):}
\begin{itemize}
    \item Hệ thống không tích hợp trực tiếp cổng thanh toán trực tuyến tự động cho khách hàng. Quy trình kiểm soát trạng thái thanh toán và kích hoạt tích điểm hoàn toàn do nhân viên quầy (Admin) xác nhận thủ công trên hệ thống sau khi khách hàng hoàn thành dịch vụ tại cửa hàng.
    \item Quy trình nâng hạ hạng được quét tự động bởi hệ thống vào ngày đầu tiên của mỗi tháng dựa trên tổng chi tiêu tích lũy của 3 tháng liền kề trước đó.
\end{itemize}

\section{Tác nhân và Sơ đồ Use Case (Actors \& Use Cases)}
Hệ thống tương tác trực tiếp với hai tác nhân chính (Primary Actors):
\begin{itemize}
    \item \textbf{Customer (Khách hàng):} Người sở hữu phương tiện, thực hiện các hành động đăng ký, quản lý danh sách biển số xe cá nhân, tra cứu thông tin phân hạng, đặt lịch hẹn rửa xe theo khung giờ khả dụng và thực hiện đổi các mã phần thưởng từ ví điểm tích lũy (\textit{Wash Credits}).
    \item \textbf{Admin (Chủ cơ sở / Nhân viên quầy):} Người vận hành hệ thống, thực hiện kiểm tra xe vào bãi, bấm xác nhận hoàn thành dịch vụ rửa xe để kích hoạt hóa đơn tích điểm, cấu hình các chương trình khuyến mãi theo hạng mục tiêu và xuất dữ liệu hành vi.
\end{itemize}

\section{Yêu cầu chức năng chi tiết (Functional Requirements)}
Các chức năng bắt buộc phải được triển khai chi tiết theo 7 Phân hệ cốt lõi (Features), thống nhất với nhãn Jira (\textit{Feature-1} đến \textit{Feature-7}) và mã thành viên phụ trách (M1--M7). Toàn bộ endpoint API sử dụng tiền tố theo đúng router thực tế của FastAPI (không có tiền tố \texttt{/api}).

\subsection{Feature 1: Đăng nhập \& Đăng ký tài khoản}
\textit{Thành viên phụ trách: M3 (Backend), M2 (Frontend)}

Phân hệ này cung cấp cơ chế xác thực an toàn cho toàn bộ người dùng của hệ thống. Bao gồm các chức năng:
\begin{enumerate}
    \item \textbf{[F1-DB] Database Schema:} Thiết kế bảng \textit{users} phục vụ lưu trữ thông tin định danh, mật khẩu đã băm và vai trò (\textit{role}) người dùng.
    \item \textbf{[F1-BE-1] API Register:} Endpoint \texttt{POST /auth/register} tiếp nhận thông tin người dùng, kiểm tra trùng lặp tên đăng nhập và thực hiện mã hóa mật khẩu bằng thuật toán Bcrypt (cost factor \(\geq\) 12) trước khi lưu trữ.
    \item \textbf{[F1-BE-2] API Login:} Endpoint \texttt{POST /auth/login} xác thực thông tin đăng nhập, sinh JSON Web Token (JWT) chứa \textit{username} và \textit{role}.
    \item \textbf{[F1-BE-3] JWT Authentication Middleware:} Middleware giải mã và xác thực JWT từ header \texttt{Authorization: Bearer <token>} để bảo vệ các endpoint riêng tư (\textit{Dependency Injection} của FastAPI).
    \item \textbf{[F1-FE-1] Giao diện đăng ký và đăng nhập:} Xây dựng biểu mẫu đăng ký, đăng nhập và lưu trữ token phía Client.
\end{enumerate}

\subsection{Feature 2: Quản lý hồ sơ xe (Vehicle Management)}
\textit{Thành viên phụ trách: M5 (Backend/Database), M2 (Frontend)}

Phân hệ cho phép khách hàng quản lý danh sách phương tiện cá nhân liên kết với tài khoản:
\begin{enumerate}
    \item \textbf{[F2-DB] Vehicle Database Schema:} Thiết kế bảng \textit{vehicles} với khóa ngoại liên kết đến bảng \textit{users}.
    \item \textbf{[F2-BE-1] CRUD Vehicle API:} Endpoint \texttt{GET/POST/PUT/DELETE /vehicles} quản lý hồ sơ xe.
    \item \textbf{[F2-BE-2] Vehicle Validation:} Kiểm tra định dạng biển số và ràng buộc quyền sở hữu (chỉ chủ xe được sửa/xóa hồ sơ của mình).
    \item \textbf{[F2-FE-1] Giao diện quản lý xe:} Màn hình thêm/sửa/xóa phương tiện và trang hồ sơ khách hàng.
\end{enumerate}

\subsection{Feature 3: Đặt lịch dịch vụ (Booking)}
\textit{Thành viên phụ trách: M3 (Backend), M2 (Frontend)}

Phân hệ xử lý toàn bộ vòng đời một lịch hẹn rửa xe:
\begin{enumerate}
    \item \textbf{[F3-DB] Booking Database Schema:} Thiết lập bảng \textit{bookings} và bảng danh mục dịch vụ \textit{services}.
    \item \textbf{[F3-BE-1] API Khởi tạo lịch hẹn:} Endpoint \texttt{POST /bookings} ràng buộc điều kiện đăng nhập, kiểm tra quy tắc cửa sổ thời gian (\textit{Booking Window}) tương ứng phân hạng thành viên trước khi thiết lập trạng thái \textit{Pending}.
    \item \textbf{[F3-BE-2] API Lịch sử \& Chi tiết lịch hẹn:} Endpoint \texttt{GET /bookings} và \texttt{GET /bookings/\{id\}} trả về danh sách/chi tiết lịch hẹn.
    \item \textbf{[F3-BE-3] API Hủy/Cập nhật lịch hẹn:} Endpoint \texttt{PUT /bookings/\{id\}} và \texttt{DELETE /bookings/\{id\}} cập nhật hoặc hủy lịch hẹn.
    \item \textbf{[F3-BE-4] Booking Priority Queue:} Logic xếp hàng ưu tiên xử lý lịch hẹn theo phân hạng thành viên (hạng cao hơn được xác nhận trước).
    \item \textbf{[F3-FE-1] Giao diện đặt lịch:} Widget chọn dịch vụ, chi nhánh, ngày giờ và biểu mẫu xác nhận thông tin khách hàng.
\end{enumerate}

\subsection{Feature 4: Thành viên \& Điểm thưởng (Loyalty Engine)}
\textit{Thành viên phụ trách: M4 (Backend), M2 (Frontend)}

Phân hệ logic phức tạp nhất hệ thống, xử lý tính điểm, thăng/hạ hạng và đổi thưởng:
\begin{enumerate}
    \item \textbf{[F4-DB] Loyalty Database Schema:} Thiết lập bảng \textit{customer\_points} (ví Wash Credits) và lịch sử biến động điểm.
    \item \textbf{[F4-BE-1] Point Calculation Engine:} Khi lịch hẹn chuyển trạng thái \textit{Completed}, hệ thống tự động cộng điểm thưởng nhân theo hệ số (\textit{Multiplier}) của hạng thành viên hiện tại.
    \item \textbf{[F4-BE-2] Tier Upgrade Engine:} Cron Job chạy 00:00 ngày đầu tháng, quét tổng chi tiêu 3 tháng gần nhất để tự động nâng/hạ hạng thành viên.
    \item \textbf{[F4-BE-3] Reward Redemption API:} Endpoint đổi điểm lấy discount/free wash/add-on, điểm tự động hết hạn sau 12 tháng.
    \item \textbf{[F4-FE-1] Loyalty Dashboard:} Giao diện hiển thị số dư điểm, hạng hiện tại và lịch sử giao dịch điểm.
\end{enumerate}

\subsection{Feature 5: Khuyến mãi \& Mã giảm giá (Promotion)}
\textit{Thành viên phụ trách: M5 (Backend/Database), M1 (Frontend Admin)}

\begin{enumerate}
    \item \textbf{[F5-DB] Promotion Database Schema:} Thiết lập bảng \textit{promotions} và quy tắc điều kiện áp dụng.
    \item \textbf{[F5-BE-1] Promotion CRUD API:} Endpoint quản lý tạo/sửa/xóa chương trình khuyến mãi.
    \item \textbf{[F5-BE-2] Target Promotion Logic:} Kiểm tra điều kiện áp dụng theo phân hạng mục tiêu (\textit{Target Tier}, ví dụ "chỉ gửi cho Silver trở lên") và khung thời gian hiệu lực.
    \item \textbf{[F5-FE-1] Giao diện quản lý khuyến mãi:} Màn hình Admin cấu hình chương trình khuyến mãi.
\end{enumerate}

\subsection{Feature 6: Quản trị \& Điều phối trung tâm (Admin Dashboard)}
\textit{Thành viên phụ trách: M3 (Backend), M1 (Frontend Admin)}

\begin{enumerate}
    \item \textbf{[F6-BE-1] Dashboard Statistics API:} Endpoint thống kê doanh thu, số lịch hẹn, hoạt động khách hàng theo thời gian thực.
    \item \textbf{[F6-BE-2] Customer Management API:} Endpoint quản lý tài khoản khách hàng cho Admin.
    \item \textbf{[F6-BE-3] Booking Management API:} Endpoint điều phối và cập nhật trạng thái lịch hẹn (Pending, Confirmed, Completed, Cancelled) cho nhân viên tại quầy.
    \item \textbf{[F6-FE-1] Bảng điều khiển quản trị:} Giao diện biểu đồ, danh sách lịch hẹn chờ xử lý dành cho người vận hành.
\end{enumerate}

\subsection{Feature 7: Khảo sát \& Nghiên cứu dữ liệu (Survey \& Research)}
\textit{Thành viên phụ trách: M7 (Backend Research API), M6 (Frontend Survey Form)}

\begin{enumerate}
    \item \textbf{[F7-BE-1] Survey API:} Endpoint tiếp nhận và lưu trữ phản hồi khảo sát khách hàng.
    \item \textbf{[F7-FE-1] Giao diện biểu mẫu khảo sát:} Form khảo sát với thang đánh giá và trường nhận xét.
    \item \textbf{[F7-BE-2] Analytics Dataset Export:} Trích xuất dữ liệu khảo sát/hành vi ra tệp phẳng \texttt{.csv} phục vụ phân tích bằng Python/Pandas hoặc Jupyter Notebook.
\end{enumerate}

\noindent \textbf{Tóm tắt số lượng công việc (Jira Tasks):} Feature 1 gồm 6 nhiệm vụ, Feature 2 gồm 5 nhiệm vụ, Feature 3 gồm 9 nhiệm vụ, Feature 4 gồm 8 nhiệm vụ, Feature 5 gồm 5 nhiệm vụ, Feature 6 gồm 4 nhiệm vụ, Feature 7 gồm 3 nhiệm vụ. Tổng cộng 40 đầu việc được kiểm soát chặt chẽ trên Jira Board dự án \textbf{KAN} thông qua hệ thống nhãn (\textit{Label}) Feature-1 đến Feature-7.

\section{Yêu cầu phi chức năng (Non-Functional Requirements)}
\begin{itemize}
    \item \textbf{Hiệu năng (Performance):} Thời gian phản hồi xử lý tính toán rẽ nhánh của các API đọc dữ liệu phải nhỏ hơn 1 giây dưới áp lực 100 người dùng truy cập đồng thời (\textit{Concurrent Users}). Tốc độ phản hồi giao diện Frontend không quá 3 giây.
    \item \textbf{Bảo mật (Security):} Toàn bộ mật khẩu lưu trữ bắt buộc đi qua bộ băm Bcrypt (\textit{cost factor} \(\geq\) 12). Phiên làm việc tương tác qua API được xác thực bằng chữ ký JSON Web Token (JWT) lưu an toàn trong thiết lập Cookie bảo mật nhằm ngăn chặn hoàn toàn lỗ hổng tấn công XSS.
    \item \textbf{Tính tin cậy và Nhất quán (Reliability):} Tác vụ xử lý tự động đầu tháng (Cron Job) phải tích hợp cơ chế \textit{Idempotency} (đảm bảo tính duy nhất), tuyệt đối không xảy ra hiện tượng chạy trùng lặp dữ liệu gây sai sót điểm số thành viên.
\end{itemize}

\section{Các quy tắc nghiệp vụ cốt lõi (Business Rules)}
Hệ thống áp dụng chặt chẽ bộ quy tắc nghiệp vụ bao gồm: Quy tắc xác thực thông tin tài khoản (\textit{Identity Rules}), Quy tắc giới hạn thời gian và hủy lịch hẹn (\textit{Booking Rules}), Quy tắc tính toán thăng hạng và hết hạn điểm (\textit{Loyalty Rules}), Quy tắc kiểm tra điều kiện áp dụng mã giảm giá (\textit{Promotion Rules}), Quy tắc bảo mật phân quyền dữ liệu (\textit{Security Rules}) và Quy tắc tách biệt dữ liệu định danh trong phân tích phục vụ nghiên cứu khoa học (\textit{Research Rules}).
