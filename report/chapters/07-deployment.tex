\section{Hệ sinh thái công nghệ và Thư viện phụ thuộc (Tech Stack)}

\subsection{Phân hệ nền tảng (Backend Stack)}
\begin{itemize}
    \item \textbf{Ngôn ngữ \& Môi trường thực thi (Runtime):} Python 3.11+.
    \item \textbf{Khung phát triển (Framework):} \textbf{FastAPI}, chạy trên ASGI server \textbf{Uvicorn}, hỗ trợ xây dựng RESTful API hiệu năng cao kèm sinh tài liệu Swagger UI tự động.
    \item \textbf{Cơ sở dữ liệu (Database):} \textbf{MySQL} -- hệ quản trị cơ sở dữ liệu quan hệ (RDBMS), bảo đảm tính toàn vẹn dữ liệu qua khóa ngoại và tối ưu hóa index truy vấn.
    \item \textbf{Trình ánh xạ dữ liệu (ORM):} \textbf{SQLAlchemy} kết hợp \textbf{Alembic} quản lý migration schema.
    \item \textbf{Bảo mật và Xác thực (Authentication):} Cơ chế Token-based sử dụng JSON Web Token (JWT, thư viện \texttt{python-jose}), kết hợp thuật toán mã hóa một chiều \texttt{bcrypt} (thư viện \texttt{passlib}) bảo vệ thông tin mật khẩu.
    \item \textbf{Bộ lập lịch tự động (Automation Service):} \textbf{APScheduler} quản trị tác vụ nền chạy ngầm (Cron Job) quét dữ liệu chi tiêu và cập nhật phân hạng tự động định kỳ đầu tháng.
    \item \textbf{Xử lý dữ liệu nghiên cứu:} \textbf{Pandas} phục vụ trích xuất, làm sạch và tổng hợp dữ liệu hành vi phục vụ Chương 5.
    \item \textbf{Kiểm thử tự động (Testing Framework):} \textbf{Pytest} kết hợp \textbf{httpx.TestClient} của FastAPI thực thi các kịch bản kiểm thử đơn vị và tích hợp.
\end{itemize}

\subsection{Phân hệ giao diện (Frontend Stack)}
\begin{itemize}
    \item \textbf{Công nghệ chính:} \textbf{HTML5, CSS3 và JavaScript thuần} (Vanilla JS) -- không sử dụng framework/thư viện biên dịch (không React, không build step).
    \item \textbf{Kết nối mạng (HTTP Client):} \texttt{Fetch API} chuẩn của trình duyệt để giao tiếp với Backend FastAPI.
    \item \textbf{Lưu trữ phiên đăng nhập:} \texttt{localStorage} lưu JWT Token phía trình duyệt sau khi đăng nhập thành công.
    \item \textbf{Giao diện trực quan (UI \& Styling):} CSS3 tùy biến thủ công, thiết kế Responsive bằng Flexbox/Grid.
    \item \textbf{Xử lý biểu mẫu dữ liệu (Form Handling):} Validation thực hiện thủ công bằng JavaScript thuần (kiểm tra định dạng, trường bắt buộc) trước khi gửi request.
\end{itemize}

\textit{Ghi chú: vì Frontend không dùng Node.js/npm, không có bước "build" -- các tệp HTML/CSS/JS được phục vụ trực tiếp (mở file hoặc dùng static file server bất kỳ).}

\section{Hướng dẫn cài đặt môi trường phát triển cục bộ (Local Setup)}
Để cấu hình và vận hành thử nghiệm hệ thống AutoWashPro tại máy tính cá nhân, thực hiện tuần tự theo các bước sau:

\subsection{Yêu cầu cài đặt tiên quyết (Prerequisites)}
\begin{enumerate}
    \item Cài đặt Python (Phiên bản $\geq$ 3.10): \url{https://www.python.org}
    \item Cài đặt hệ quản trị cơ sở dữ liệu MySQL (Phiên bản $\geq$ 8.0): \url{https://www.mysql.com}
    \item Cài đặt công cụ quản lý mã nguồn Git: \url{https://git-scm.com}
\end{enumerate}

\subsection{Các bước triển khai Backend}
\begin{enumerate}
    \item \textbf{Tải mã nguồn từ kho lưu trữ (Clone Repository):}
\begin{verbatim}
git clone https://github.com/<tai-khoan-nhom>/SWP391_AutoWashPro
cd SWP391_AutoWashPro/backend
\end{verbatim}

    \item \textbf{Tạo và kích hoạt môi trường ảo Python (Virtual Environment):}
\begin{verbatim}
python -m venv .venv
# Windows:
.venv\Scripts\Activate.ps1
# macOS/Linux:
source .venv/bin/activate
\end{verbatim}

    \item \textbf{Cài đặt thư viện phụ thuộc:}
\begin{verbatim}
pip install -r requirement.txt
\end{verbatim}

    \item \textbf{Khởi chạy máy chủ API (ASGI Server):}
\begin{verbatim}
uvicorn app.main:app --reload
# Server chạy tại http://127.0.0.1:8000
# Swagger UI: http://127.0.0.1:8000/docs
\end{verbatim}
\end{enumerate}

\textit{Ghi chú: tại thời điểm biên soạn báo cáo, hệ thống chưa kết nối MySQL thực tế (dữ liệu Auth/Booking đang lưu tạm trong bộ nhớ RAM của tiến trình Python để phục vụ phát triển nhanh) và chưa sử dụng tệp cấu hình \texttt{.env} -- đây là hạng mục cần hoàn thiện tiếp theo (kết nối \texttt{database.py} với SQLAlchemy, di chuyển \texttt{SECRET\_KEY} ra biến môi trường thay vì hard-code trong mã nguồn).}

\subsection{Các bước triển khai Frontend}
\begin{enumerate}
    \item Không cần cài đặt phụ thuộc (không dùng npm/Node.js).
    \item Mở trực tiếp tệp \texttt{frontend/index.html} bằng trình duyệt, hoặc phục vụ qua static server bất kỳ để tránh giới hạn CORS khi test:
\begin{verbatim}
cd frontend
python -m http.server 5500
# Truy cập http://127.0.0.1:5500
\end{verbatim}
    \item Đảm bảo Backend đã bật \texttt{CORSMiddleware} (xem Chương 4) để Frontend gọi API cross-origin thành công.
\end{enumerate}

\section{Triển khai hệ thống lên môi trường Production}

\subsection{Triển khai API Server lên nền tảng đám mây}
Backend FastAPI có thể triển khai lên các nền tảng PaaS hỗ trợ Python (ví dụ: Render, Railway, PythonAnywhere) thông qua liên kết trực tiếp với nhánh \texttt{main} của GitHub. Lệnh khởi chạy production khuyến nghị sử dụng Gunicorn quản lý worker cho Uvicorn:
\begin{verbatim}
gunicorn app.main:app -k uvicorn.workers.UvicornWorker --bind 0.0.0.0:$PORT
\end{verbatim}

\subsection{Triển khai Giao diện Client lên dịch vụ Hosting tĩnh}
Vì Frontend là các tệp HTML/CSS/JS thuần, có thể triển khai trực tiếp (không cần build) lên các dịch vụ hosting tĩnh như Netlify, Vercel hoặc GitHub Pages bằng cách trỏ thẳng vào thư mục \texttt{frontend/}.

\subsection{Quy trình sao lưu dữ liệu hệ thống (Database Backup)}
Khi đã kết nối MySQL thực tế, định kỳ thực hiện sao lưu dữ liệu:
\begin{verbatim}
# Sao lưu dữ liệu sang tệp backup SQL
mysqldump -u root -p autowashpro > backup_database.sql

# Khôi phục dữ liệu tại máy chủ production mới
mysql -u root -p autowashpro < backup_database.sql
\end{verbatim}

\section{Tài liệu hướng dẫn vận hành hệ thống (User Manual)}

\subsection{Cẩm nang dành cho Khách hàng sử dụng ứng dụng (Client Menu)}

\subsubsection{Bước 1: Khởi tạo tài khoản hệ thống (Đã triển khai)}
\begin{enumerate}
    \item Khách hàng truy cập trang chủ ứng dụng, chọn mục "Đăng ký tài khoản".
    \item Nhập Tên đăng nhập và Mật khẩu.
    \item Sử dụng thông tin vừa đăng ký để "Đăng nhập", hệ thống trả về JWT Token lưu tại \texttt{localStorage} của trình duyệt.
\end{enumerate}

\subsubsection{Bước 2: Cập nhật hồ sơ phương tiện cá nhân (Kế hoạch -- Feature 2 chưa triển khai)}
\begin{enumerate}
    \item Di chuyển đến khu vực quản lý cá nhân, chọn menu "Hồ sơ xe của tôi".
    \item Bấm "Thêm phương tiện mới", điền Hãng xe, Dòng xe, Biển số và Màu sắc.
\end{enumerate}

\subsubsection{Bước 3: Thiết lập đặt lịch hẹn dịch vụ trực tuyến (Đã triển khai một phần)}
\begin{enumerate}
    \item Tại màn hình chính, chọn tính năng "Đặt lịch dịch vụ".
    \item Chọn gói dịch vụ mong muốn và điền thông tin ngày hẹn.
    \item Bấm "Xác nhận lịch đặt" -- hệ thống gọi \texttt{POST /bookings}, khởi tạo bản ghi với trạng thái \texttt{Pending}.
    \item \textit{Lưu ý: quy tắc Booking Window theo hạng thành viên và kiểm tra khung giờ trống (Slot) hiện chưa được lập trình -- xem Chương 4, mục Kế hoạch triển khai.}
\end{enumerate}

\subsubsection{Bước 4: Quản lý Ví điểm thưởng Loyalty và Áp dụng mã giảm giá (Kế hoạch -- Feature 4, 5 chưa triển khai)}
\begin{enumerate}
    \item Khách hàng truy cập menu "Ví thành viên" để kiểm tra số dư điểm thưởng (\textit{Wash Credits}) và phân hạng hiện tại.
    \item Đổi điểm lấy mã ưu đãi, áp dụng mã giảm giá khi đặt lịch.
\end{enumerate}

\subsubsection{Bước 5: Tra cứu lịch sử \& Hủy lịch (Đã triển khai)}
\begin{enumerate}
    \item Khách hàng vào mục "Lịch sử dịch vụ" (\texttt{GET /bookings}) để theo dõi các lịch hẹn đã đặt.
    \item Bấm "Hủy lịch hẹn" (\texttt{DELETE /bookings/\{id\}}) để hủy lịch. \textit{Quy tắc giới hạn thời gian hủy (>6 tiếng) và hoàn điểm tự động thuộc phạm vi Kế hoạch, chưa được lập trình.}
\end{enumerate}

\subsection{Cẩm nang dành cho Nhân viên và Quản trị viên (Admin/Staff Dashboard)}
\textit{Toàn bộ mục này thuộc Feature 6 (Admin Dashboard) -- hiện đang trong giai đoạn triển khai (router \texttt{admin/} chưa hoàn thiện mã nguồn). Nội dung dưới đây mô tả thiết kế nghiệp vụ mục tiêu.}

\subsubsection{Bước 1: Truy cập trang điều hành tập trung (Kế hoạch)}
\begin{enumerate}
    \item Nhân viên truy cập đường dẫn phân hệ quản trị dành riêng (ví dụ: \texttt{/admin} phía Frontend).
    \item Đăng nhập bằng tài khoản có \texttt{role = "Admin"}, hệ thống điều hướng vào Admin Dashboard.
\end{enumerate}

\subsubsection{Bước 2: Điều phối lịch hẹn tại quầy (Kế hoạch)}
\begin{enumerate}
    \item Nhân viên xem danh sách lịch hẹn trong ngày, cập nhật trạng thái (Pending $\rightarrow$ Confirmed $\rightarrow$ Completed).
    \item Khi chuyển trạng thái Completed, hệ thống Loyalty (Feature 4) tự động cộng điểm thưởng nhân theo hệ số hạng của khách hàng.
\end{enumerate}

\subsubsection{Bước 3: Quản lý danh mục cấu hình và Theo dõi dữ liệu nghiên cứu (Kế hoạch)}
\begin{enumerate}
    \item Quản trị viên quản lý bảng đơn giá dịch vụ và các chương trình khuyến mãi (Feature 5).
    \item Truy cập "Dữ liệu nghiên cứu" để xuất dữ liệu hành vi (Feature 7) sang tệp \texttt{.csv} ẩn danh phục vụ Chương 5.
\end{enumerate}

\section{Kết luận và Định hướng phát triển tương lai}
Dự án Quản lý Rửa xe Thông minh (\textbf{AutoWashPro}) đã hoàn thành phân hệ lõi Feature 1 (Xác thực) và Feature 3 (Đặt lịch dịch vụ) với API hoạt động ổn định trên nền tảng FastAPI, cùng giao diện Frontend HTML/CSS/JS kết nối trực tiếp qua Fetch API. Các phân hệ còn lại (Quản lý xe, Loyalty, Khuyến mãi, Admin Dashboard, Khảo sát \& Nghiên cứu) đã được đặc tả chi tiết tại Chương 3 và 4, hiện đang trong giai đoạn triển khai theo tiến độ Sprint trên Jira Board.

\noindent \textbf{Các hạng mục cần hoàn thiện trước khi bàn giao chính thức:}
\begin{itemize}
    \item Kết nối cơ sở dữ liệu MySQL thực tế qua SQLAlchemy (thay thế lưu trữ tạm trong RAM), triển khai băm mật khẩu Bcrypt thay vì lưu plaintext.
    \item Hoàn thiện middleware xác thực JWT (\texttt{dependencies.py}) để bảo vệ các endpoint riêng tư.
    \item Lập trình đầy đủ 5 phân hệ còn lại: Vehicle Management, Loyalty Engine, Promotion, Admin Dashboard, Survey \& Research.
    \item Viết bộ kiểm thử tự động (Pytest) đạt độ bao phủ mục tiêu.
\end{itemize}

\noindent \textbf{Các hướng nghiên cứu và phát triển mở rộng trong tương lai:}
\begin{itemize}
    \item Ứng dụng thuật toán học máy (Machine Learning Recommendation Engine) phân tích tệp dữ liệu log nghiên cứu \texttt{.csv} để tự động gợi ý gói dịch vụ cá nhân hóa và dự báo hành vi rời bỏ hệ thống của khách hàng.
    \item Đóng gói và phát triển phiên bản ứng dụng di động (Mobile Application) trên iOS và Android.
    \item Tích hợp mã QR Code động tại quầy giúp khách hàng tự động check-in xác nhận xe vào bãi.
    \item Nâng cấp kiến trúc thông báo thời gian thực (WebSocket) để đẩy thông tin tiến độ rửa xe trực tiếp đến khách hàng.
\end{itemize}
