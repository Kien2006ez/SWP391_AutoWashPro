\section{Tech Stack \& Dependencies}

\subsection{Backend Stack}
\begin{itemize}
    \item \textbf{Runtime:} Node.js 20+ hoặc Python 3.11+
    \item \textbf{Framework:} Express.js (Node) hoặc Flask (Python)
    \item \textbf{Database:} PostgreSQL 15+ (RDBMS chính)
    \item \textbf{Authentication:} JWT (JSON Web Token), bcrypt password hashing
    \item \textbf{Payment Gateway Integration:} VNPay API / Stripe API / Momo API (tùy chọn)
    \item \textbf{Email Service:} SendGrid hoặc Nodemailer
    \item \textbf{ORM/Query Builder:} Sequelize (Node) hoặc SQLAlchemy (Python)
    \item \textbf{Validation:} Joi / Yup (Node) hoặc Marshmallow (Python)
    \item \textbf{Testing:} Jest / Mocha (Node) hoặc Pytest (Python)
    \item \textbf{CI/CD:} GitHub Actions, Docker containerization
\end{itemize}

\subsection{Frontend Stack}
\begin{itemize}
    \item \textbf{Framework:} React 18+, Next.js 13+ (optional)
    \item \textbf{State Management:} Redux Toolkit hoặc Zustand
    \item \textbf{HTTP Client:} Axios / Fetch API
    \item \textbf{UI Component Library:} Material-UI (MUI), Tailwind CSS
    \item \textbf{Form Handling:} React Hook Form, Formik
    \item \textbf{Testing:} Jest, React Testing Library
    \item \textbf{Deployment:} Netlify / Vercel
\end{itemize}

\section{Installation Guide - Local Development Environment}

Để cấu hình và khởi chạy hệ thống AutoWashPro tại môi trường máy tính cục bộ, thực hiện các bước sau:

\subsection{Prerequisites}
\begin{enumerate}
    \item Cài đặt Node.js 20+ (hoặc Python 3.11+): \url{https://nodejs.org}
    \item Cài đặt PostgreSQL 15+: \url{https://www.postgresql.org/download}
    \item Cài đặt Git: \url{https://git-scm.com}
    \item (Optional) Cài đặt Docker: \url{https://www.docker.com}
\end{enumerate}

\subsection{Step-by-Step Setup}

\begin{enumerate}
    \item \textbf{Clone Repository:}
    \begin{verbatim}
git clone https://github.com/kienlndev/SWP391_AutoWashPro.git
cd SWP391_AutoWashPro
    \end{verbatim}

    \item \textbf{Cấu hình Environment Variables:}
    \begin{verbatim}
cp .env.example .env
# Chỉnh sửa .env với thông số:
# DATABASE_URL=postgresql://user:password@localhost:5432/autowashpro
# JWT_SECRET=your_jwt_secret_key_here
# VNPAY_MERCHANT_ID=your_vnpay_merchant_id
# MAIL_SERVICE=sendgrid (hoặc nodemailer)
    \end{verbatim}

    \item \textbf{Cài đặt Dependencies (Backend):}
    \begin{verbatim}
cd backend
npm install
# hoặc nếu dùng Python:
pip install -r requirements.txt
    \end{verbatim}

    \item \textbf{Khởi tạo Database:}
    \begin{verbatim}
npm run db:migrate
# hoặc
python manage.py db upgrade
    \end{verbatim}

    \item \textbf{Cài đặt Dependencies (Frontend):}
    \begin{verbatim}
cd ../frontend
npm install
    \end{verbatim}

    \item \textbf{Khởi chạy Development Servers:}
    \begin{verbatim}
# Terminal 1 - Backend
cd backend
npm run dev
# Runs on http://localhost:5000

# Terminal 2 - Frontend
cd frontend
npm start
# Runs on http://localhost:3000
    \end{verbatim}

    \item \textbf{(Optional) Docker Compose:}
    \begin{verbatim}
docker-compose up -d
# Automatically runs both backend (5000) and frontend (3000)
    \end{verbatim}
\end{enumerate}

\section{Deployment to Production}

\subsection{Backend Deployment (Heroku)}
\begin{verbatim}
heroku login
heroku create your-app-name
git push heroku main
heroku run npm run db:migrate
heroku open
\end{verbatim}

\subsection{Frontend Deployment (Netlify)}
\begin{verbatim}
npm run build
# Connect to Netlify via Git integration
# Auto-deploy on main branch push
\end{verbatim}

\subsection{Database Backup \& Migration}
\begin{verbatim}
pg_dump autowashpro > backup.sql
# Transfer to production server
psql autowashpro < backup.sql
\end{verbatim}

\section{User Manual}

\subsection{Hướng dẫn dành cho Khách hàng (Customer Manual)}

\subsubsection{Bước 1: Đăng ký tài khoản}
\begin{enumerate}
    \item Truy cập trang chủ, click "Đăng ký"
    \item Nhập email, số điện thoại, mật khẩu (tối thiểu 8 ký tự)
    \item Nhận mã OTP qua SMS, nhập mã 6 ký tự
    \item Tài khoản được kích hoạt, click "Đăng nhập"
\end{enumerate}

\subsubsection{Bước 2: Quản lý xe cá nhân}
\begin{enumerate}
    \item Đi tới "Danh sách xe" trong menu
    \item Click "Thêm xe mới"
    \item Nhập hãng, model, biển số, màu sắc
    \item Lưu lại
\end{enumerate}

\subsubsection{Bước 3: Đặt lịch rửa xe}
\begin{enumerate}
    \item Click "Đặt lịch rửa xe"
    \item Chọn xe từ danh sách
    \item Chọn dịch vụ (Rửa nhanh/chi tiết/Nano)
    \item Chọn ngày trên calendar (max 30 ngày sau)
    \item Chọn khung giờ trống (ví dụ: 09:00-10:00)
    \item Review thông tin, click "Xác nhận"
\end{enumerate}

\subsubsection{Bước 4: Thanh toán}
\begin{enumerate}
    \item Sau khi booking được confirm, hóa đơn tự động tạo
    \item Vào "Hóa đơn của tôi", click vào hóa đơn
    \item Click "Thanh toán"
    \item Chọn phương thức: Thẻ tín dụng / Chuyển khoản / Mobile wallet
    \item Nhập thông tin, hoàn tất giao dịch
    \item Nhận email xác nhận thanh toán
\end{enumerate}

\subsubsection{Bước 5: Xem lịch sử \& Hủy booking}
\begin{enumerate}
    \item Menu "Lịch sử đặt lịch"
    \item Xem chi tiết: xe, dịch vụ, ngày/giờ, giá
    \item Nếu cần hủy (còn >6 giờ), click "Hủy"
    \item Hoàn tiền sẽ được xử lý trong 3-5 ngày
\end{enumerate}

\subsection{Hướng dẫn dành cho Quản lý (Admin Manual)}

\subsubsection{Bước 1: Đăng nhập Admin Dashboard}
\begin{enumerate}
    \item Truy cập \url{https://autowashpro-demo.netlify.app/admin}
    \item Nhập email admin (setup sẵn), mật khẩu
    \item Vào Dashboard chính
\end{enumerate}

\subsubsection{Bước 2: Quản lý lịch hẹn trong ngày}
\begin{enumerate}
    \item Menu "Lịch hẹn ngày hôm nay"
    \item Xem danh sách khách hàng, biển số xe, giờ dự kiến
    \item Click "Tiếp nhận xe" khi khách đến
    \item Update status: Đang rửa → Hoàn thành (tự động tích điểm)
\end{enumerate}

\subsubsection{Bước 3: Xem thống kê doanh thu}
\begin{enumerate}
    \item Menu "Dashboard / Thống kê"
    \item Xem biểu đồ: Doanh thu theo ngày/tháng, booking rate, customer retention
    \item Export report thành CSV
\end{enumerate}

\subsubsection{Bước 4: Quản lý dịch vụ \& giá}
\begin{enumerate}
    \item Menu "Dịch vụ"
    \item Click "Thêm dịch vụ mới" hoặc edit dịch vụ hiện tại
    \item Cập nhật tên, giá, thời gian, mô tả
    \item Lưu lại
\end{enumerate}

\subsubsection{Bước 5: Xem danh sách khách hàng \& phân hạng}
\begin{enumerate}
    \item Menu "Khách hàng"
    \item Tìm kiếm theo số điện thoại hoặc biển số
    \item Xem hạng thành viên (Member/Silver/Gold/Platinum)
    \item Xem lịch sử mua hàng, điểm tích lũy
\end{enumerate}
