\chapter{Kiểm Thử Hệ Thống AutoWashPro}

\section{Kế hoạch \& Chiến lược kiểm thử}
Quá trình kiểm thử hệ thống AutoWashPro được thực hiện nghiêm ngặt theo các cấp độ kiến trúc: Unit Tests (kiểm thử đơn vị từng phân hệ API), Integration Tests (kiểm thử luồng liên kết nghiệp vụ), System Tests (kiểm thử toàn bộ hệ thống đầu cuối) và User Acceptance Tests (kiểm thử hộp đen với người dùng thực tế). Toàn bộ kịch bản kiểm thử (Test Cases) được quản trị tập trung trên Jira Board dự án \textbf{KAN} qua nhãn (Label) \texttt{Testing} kết hợp \texttt{Feature-1} đến \texttt{Feature-7}.

\textit{Ghi chú quan trọng: tại thời điểm biên soạn báo cáo, chỉ Feature 1 (Auth) và Feature 3 (Booking) đã có mã nguồn API để kiểm thử thực tế. Các Feature 2, 4, 5, 6, 7 hiện đang trong giai đoạn triển khai (router trống) -- test case tương ứng được thiết kế trước theo đặc tả Chương 3 và sẽ được thực thi khi mã nguồn hoàn thiện.}

\section{Integration Tests (Kiểm thử tích hợp chu trình đầu cuối)}

\subsection{End-to-End Flow Tests}

\begin{enumerate}
    \item \textbf{TC-INT-01: Quy trình Đăng ký \& Đăng nhập Toàn diện (Đã triển khai)}
    \begin{itemize}
        \item Khách hàng gửi yêu cầu đăng ký tài khoản mới $\rightarrow$ Đăng nhập bằng đúng cặp tài khoản/mật khẩu vừa tạo $\rightarrow$ Nhận JWT Token.
        \item Kết quả mong đợi: Hệ thống phản hồi JWT Token hợp lệ, \texttt{username} và \texttt{role} chính xác.
    \end{itemize}

    \item \textbf{TC-INT-02: Chu trình Đặt lịch dịch vụ cơ bản (Đã triển khai một phần)}
    \begin{itemize}
        \item Khách hàng gọi \texttt{POST /bookings} với thông tin dịch vụ và ngày hẹn $\rightarrow$ Tra cứu lại bằng \texttt{GET /bookings/\{id\}}.
        \item Kết quả mong đợi: Bản ghi Booking được lưu với trạng thái ban đầu là Pending và có thể truy vấn lại chính xác.
    \end{itemize}

    \item \textbf{TC-INT-03: Luồng Hoàn thành dịch vụ, Tích điểm \& Trích xuất Nhật ký nghiên cứu (Kế hoạch)}
    \begin{itemize}
        \item Nhân viên tại quầy xác nhận booking chuyển trạng thái Completed $\rightarrow$ Hệ thống Loyalty tính toán điểm thưởng nhân theo hệ số hạng $\rightarrow$ Triển khai lưu log hành vi.
        \item Kết quả mong đợi: Số dư ví điểm \textit{Wash Credits} của khách hàng tăng đúng theo tỷ lệ hạng, bộ \textit{Research Logger} xuất dữ liệu dạng phẳng (.csv) ghi nhận đầy đủ các trường thông tin ẩn danh phục vụ bài toán nghiên cứu khoa học ở Chương 5.
    \end{itemize}

    \item \textbf{TC-INT-04: Luồng Hủy lịch hẹn hoàn điểm (Kế hoạch)}
    \begin{itemize}
        \item Khách hàng thực hiện hủy lịch đã đặt bằng hình thức quy đổi điểm trước đó.
        \item Kết quả mong đợi: Trạng thái lịch hẹn chuyển sang Cancelled, hệ thống hoàn trả lại số điểm thưởng tương ứng.
    \end{itemize}
\end{enumerate}
