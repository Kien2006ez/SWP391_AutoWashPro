\chapter{Kiểm Thử Hệ Thống AutoWashPro}

\section{Kế hoạch \& Chiến lược kiểm thử}
Quá trình kiểm thử hệ thống AutoWashPro được thực hiện nghiêm ngặt theo các cấp độ kiến trúc: Unit Tests (kiểm thử đơn vị từng phân hệ API), Integration Tests (kiểm thử luồng liên kết nghiệp vụ), System Tests (kiểm thử toàn bộ hệ thống đầu cuối) và User Acceptance Tests (kiểm thử hộp đen với người dùng thực tế). Toàn bộ 51 kịch bản kiểm thử (Test Cases) được quản trị tập trung trên Jira board qua dải mã định danh từ KAN-92 đến KAN-95 vàđược quản lý tập trung trên Jira Kanban Board và được thực hiện
xuyên suốt quá trình phát triển dự án.

\section{Integration Tests (Kiểm thử tích hợp chu trình đầu cuối)}

\subsection{End-to-End Flow Tests}

\begin{enumerate}
    \item \textbf{TC-INT-01: Quy trình Xác thực \& Kích hoạt Tài khoản Toàn diện}
    \begin{itemize}
        \item Khách hàng gửi yêu cầu đăng ký tài khoản mới $\rightarrow$ Kiểm tra bản ghi sinh ra trong Database ở trạng thái Pending $\rightarrow$ Nhập mã OTP chính xác gửi qua hệ thống $\rightarrow$ Đăng nhập bằng cặp tài khoản mật khẩu vừa tạo.
        \item Kết quả mong đợi: Trạng thái tài khoản chuyển thành Active, hệ thống phản hồi JWT Token hợp lệ và điều hướng thành công vào màn hình Dashboard cá nhân.
    \end{itemize}
    
    \item \textbf{TC-INT-02: Chu trình Đặt lịch dịch vụ thời gian thực}
    \begin{itemize}
        \item Khách hàng thêm mới phương tiện $\rightarrow$ Truy vấn danh mục gói dịch vụ $\rightarrow$ Chọn khung giờ slot trống $\rightarrow$ Xác nhận đặt lịch hẹn (Booking).
        \item Kết quả mong đợi: Bản ghi Booking được lưu với trạng thái ban đầu là Pending, hệ thống tự động khóa giữ slot thời gian đã chọn để tránh ghi đè dữ liệu, gửi email xác nhận.
    \end{itemize}
    
    \item \textbf{TC-INT-03: Luồng Hoàn thành dịch vụ, Tích điểm \& Trích xuất Nhật ký nghiên cứu}
    \begin{itemize}
        \item Nhân viên tại quầy xác nhận xe đã hoàn thành rửa thực tế $\rightarrow$ Hệ thống cập nhật trạng thái lịch hẹn thành Completed $\rightarrow$ Tự động sinh hóa đơn nội bộ $\rightarrow$ Hệ thống Loyalty tính toán điểm thưởng nhân theo hệ số hạng $\rightarrow$ Triển khai lưu log hành vi.
        \item Kết quả mong đợi: Hóa đơn nội bộ chuyển trạng thái hoàn tất, số dư ví điểm \textit{Wash Credits} của khách hàng tăng lên chính xác theo tỷ lệ hạng, bộ \textit{Research Logger} xuất dữ liệu dạng phẳng (.csv) ghi nhận đầy đủ 10 trường thông tin ẩn danh phục vụ bài toán nghiên cứu khoa học ở Chương 5.
    \end{itemize}

    \item \textbf{TC-INT-04: Luồng Hủy lịch hẹn đổi mã ưu đãi hoàn điểm}
    \begin{itemize}
        \item Khách hàng thực hiện hủy lịch đặt trước khung giờ thi công quy định > 6 tiếng.
        \item Kết quả mong đợi: Trạng thái lịch hẹn chuyển sang Cancelled, hệ thống tự động giải phóng khung giờ trống trên lịch trực tuyến, hoàn trả lại số điểm thưởng tương ứng nếu lịch hẹn đó được đặt bằng hình thức quy đổi điểm trước đó.
    \end{itemize}
\end{enumerate}
