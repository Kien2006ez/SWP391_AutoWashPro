\chapter{Thiết Kế Kiến Trúc Và Hệ Thống}

Nhóm tiến hành phân tích hệ thống dữ liệu cốt lõi, kiến trúc mã nguồn tổng quan và giao diện người dùng dựa trên các yêu cầu chức năng của hệ thống Quản lý Rửa xe Thông minh (\textbf{AutoWashPro}). Toàn bộ quy trình thiết kế được bám sát theo tiến độ phân rã các thẻ công việc trên bảng quản trị Jira.

\section{Thiết kế Cơ sở dữ liệu và Sơ đồ ERD (KAN-40)}
Hệ thống sử dụng cơ sở dữ liệu quan hệ nhằm đảm bảo tính toàn vẹn dữ liệu, tính nhất quán trong các luồng nghiệp vụ phức tạp như xác thực tài khoản, đặt lịch hẹn và xử lý hóa đơn thanh toán.

\subsection{Sơ đồ mối quan hệ thực thể (ERD Diagram)}
Sơ đồ ERD dưới đây biểu diễn mối quan hệ logic giữa các thực thể chính trong hệ thống bao gồm người dùng (\textit{Users}), phương tiện xe (\textit{Vehicles}), lịch hẹn (\textit{Bookings}), hóa đơn (\textit{Invoices}) và danh mục dịch vụ (\textit{Services}).

\subsection{Đặc tả cấu trúc chi tiết các bảng dữ liệu (Database Schema)}
Dưới đây là chi tiết cấu trúc bảng thuộc tính phục vụ trực tiếp cho Phân hệ Xác thực và Đăng ký người dùng (\textbf{Feature 1}).

\subsubsection{Bảng dữ liệu Người dùng (Users Table)}
Bảng \textbf{Users} lưu trữ thông tin tài khoản của toàn bộ hệ thống bao gồm Khách hàng, Nhân viên và Quản trị viên (Admin).

\begin{table}[htbp]
\centering
\caption{Đặc tả chi tiết thuộc tính bảng Users}
\label{tab:users_schema}
\small
\begin{tabular}{llcl}
\toprule
\textbf{Tên Trường} & \textbf{Kiểu Dữ Liệu} & \textbf{Khóa} & \textbf{Mô Tả Chức Năng} \\
\midrule
id & INT (AUTO\_INCREMENT) & PK & Mã định danh duy nhất của người dùng \\
email & VARCHAR(100) & Unique & Địa chỉ email dùng để đăng nhập hệ thống \\
phone & VARCHAR(15) & Unique & Số điện thoại nhận mã xác thực OTP \\
password\_hash & VARCHAR(255) & - & Mật khẩu đã được mã hóa an toàn bằng bcrypt \\
fullname & VARCHAR(100) & - & Họ và tên đầy đủ của người dùng \\
role & ENUM('Customer', 'Staff', 'Admin') & - & Vai trò phân quyền tương ứng trong hệ thống \\
status & ENUM('Pending', 'Active', 'Suspended') & - & Trạng thái hoạt động của tài khoản \\
created\_at & TIMESTAMP & - & Thời gian khởi tạo tài khoản hệ thống \\
\bottomrule
\end{tabular}
\end{table}

\subsubsection{Bảng dữ liệu Xác thực OTP (OTP\_Verifications Table)}
Bảng này lưu trữ tạm thời các mã OTP phục vụ luồng xác thực kích hoạt tài khoản an toàn với thời gian hết hạn cố định là 5 phút.

\begin{table}[htbp]
\centering
\caption{Đặc tả chi tiết thuộc tính bảng OTP\_Verifications}
\label{tab:otp_schema}
\small
\begin{tabular}{llcl}
\toprule
\textbf{Tên Trường} & \textbf{Kiểu Dữ Liệu} & \textbf{Khóa} & \textbf{Mô Tả Chức Năng} \\
\midrule
id & INT (AUTO\_INCREMENT) & PK & Mã định danh bản ghi xác thực \\
user\_id & INT & FK & Liên kết tới bảng Users (Xác định chủ tài khoản) \\
otp\_code & VARCHAR(6) & - & Mã số xác thực ngẫu nhiên gồm 6 ký tự \\
expires\_at & TIMESTAMP & - & Thời điểm mã số OTP hết hiệu lực (5 phút) \\
is\_used & BOOLEAN & - & Đánh dấu mã đã được sử dụng hay chưa \\
\bottomrule
\end{tabular}
\end{table}

\section{Kiến trúc và Cấu trúc thư mục mã nguồn Backend (KAN-21, KAN-24)}
Để đảm bảo hệ thống dễ dàng bảo trì, mở rộng và phát triển song song giữa các thành viên, cấu trúc mã nguồn Backend được phân chia theo mô hình cấu trúc phân lớp rõ ràng, quản lý chặt chẽ trên kho lưu trữ GitHub.

\begin{verbatim}
SWP391_AutoWashPro/
├── database/                # Nơi lưu trữ các file kịch bản SQL khởi tạo cơ sở dữ liệu

├── app/                     # Thư mục mã nguồn chính của ứng dụng ứng dụng Flask API
│   ├── __init__.py          # Khởi tạo cấu hình ứng dụng, định cấu hình môi trường và các gói mở rộng
│   ├── routes/              # Xử lý các chức năng phân hệ phía người dùng (Client-side Endpoints)
│   │   ├── index.py         # Trang chủ hiển thị thông tin và điều hướng hệ thống tổng quan
│   │   ├── auth.py          # Chức năng Đăng nhập, Đăng ký, Xác thực OTP (KAN-41, KAN-43)
│   │   └── booking.py       # Chức năng Đặt lịch rửa xe và xử lý thanh toán trực tuyến
│   ├── admin/               # Phân hệ quản trị dành cho chủ cửa hàng/nhân viên điều phối
│   │   ├── dashboard.py     # Trang tổng quan, thống kê doanh thu và quản trị số lượng xe
│   │   ├── services.py      # Quản lý các gói dịch vụ rửa xe và thiết lập tích lũy điểm
│   │   └── orders.py        # Tiếp nhận và cập nhật trạng thái lịch hẹn rửa xe theo thời gian thực
│   └── static/              # Quản lý các tài nguyên tĩnh hệ thống (Assets)
│       ├── css/             # Các file cấu hình giao diện, định dạng giao diện ứng dụng
│       ├── js/              # Thư viện Javascript xử lý bất đồng bộ và tương tác giao diện
│       └── images/          # Hình ảnh minh họa, biểu tượng logo thương hiệu hệ thống
├── templates/               # Chứa các tệp tin giao diện HTML ứng dụng phía máy chủ
├── config.py                # Cấu hình môi trường và thông số kết nối Database mật mã (KAN-24)
├── run.py                   # Tệp tin kích hoạt khởi chạy toàn bộ hệ thống cục bộ
├── .gitignore               # Loại bỏ các tệp không cần thiết (venv, cache) khỏi kho Git
└── requirements.txt         # Danh sách thư viện phụ thuộc bắt buộc (Flask, PyMySQL, PyJWT)

\section{Thiết kế API REST Specification (KAN-41, KAN-42, KAN-43)}
Hệ thống sử dụng kiến trúc RESTful API phía Backend tương tác với Frontend thông qua các endpoint HTTP chuẩn. Dưới đây là đặc tả chi tiết các API endpoints cho Feature 1 (Authentication):

\subsection{API Authentication Endpoints}

\subsubsection{POST /api/auth/register - Đăng ký tài khoản mới}
\begin{verbatim}
Request Body:
{
  "email": "user@example.com",
  "phone": "0987654321",
  "password": "SecurePass123",
  "fullname": "Nguyễn Văn A"
}

Response (201 Created):
{
  "status": "success",
  "message": "OTP đã được gửi đến email/SMS",
  "user_id": 123,
  "expires_at": "2026-06-10T15:05:00Z"
}

Error Response (400 Bad Request):
{
  "status": "error",
  "message": "Email đã tồn tại",
  "errors": { "email": "Email này đã được đăng ký" }
}
\end{verbatim}

\subsubsection{POST /api/auth/verify-otp - Xác thực mã OTP}
\begin{verbatim}
Request Body:
{
  "email": "user@example.com",
  "otp_code": "123456"
}

Response (200 OK):
{
  "status": "success",
  "message": "Tài khoản đã được kích hoạt",
  "user": { "id": 123, "email": "user@example.com", "status": "Active" }
}

Error Response (400 Bad Request):
{
  "status": "error",
  "message": "Mã OTP không hợp lệ hoặc đã hết hạn"
}
\end{verbatim}

\subsubsection{POST /api/auth/login - Đăng nhập}
\begin{verbatim}
Request Body:
{
  "email": "user@example.com",
  "password": "SecurePass123"
}

Response (200 OK):
{
  "status": "success",
  "access_token": "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...",
  "refresh_token": "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...",
  "user": {
    "id": 123,
    "email": "user@example.com",
    "fullname": "Nguyễn Văn A",
    "role": "Customer"
  }
}

Error Response (401 Unauthorized):
{
  "status": "error",
  "message": "Email hoặc mật khẩu không chính xác"
}
\end{verbatim}

\subsubsection{POST /api/auth/refresh-token - Làm mới Access Token}
\begin{verbatim}
Request Header:
Authorization: Bearer <refresh_token>

Response (200 OK):
{
  "status": "success",
  "access_token": "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9..."
}

Error Response (401 Unauthorized):
{
  "status": "error",
  "message": "Token không hợp lệ hoặc đã hết hạn"
}
\end{verbatim}
